create table SAQUE (
  SAQUE_ID                   INT AUTO_INCREMENT PRIMARY KEY,
  VALOR_SAQUE                FLOAT NOT NULL,
  DATA_SAQUE                 DATE NOT NULL,
  OBSERVACAO_SAQUE           VARCHAR(50) NULL
);


/* Checagem */
alter table SAQUE
add constraint CK_VALOR_SAQUE
check (VALOR_SAQUE > 0);
