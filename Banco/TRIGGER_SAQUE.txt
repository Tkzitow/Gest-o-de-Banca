DELIMITER //

create trigger TGR_SAQUE_INSERT
after insert
on SAQUE
for each row
begin
  update FINANCEIRO
  set BANCA_ATUAL = BANCA_ATUAL - new.VALOR_SAQUE,
  ROI = BANCA_ATUAL - BANCA;
end //

DELIMITER ;

DELIMITER //

create trigger TGR_SAQUE_UPDATE
after update
on SAQUE
for each row
begin
  update FINANCEIRO
  set BANCA_ATUAL = BANCA_ATUAL - new.VALOR_SAQUE,
      ROI = BANCA_ATUAL - BANCA;
end //

DELIMITER ;
